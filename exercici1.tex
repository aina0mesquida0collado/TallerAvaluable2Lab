\section{\texorpdfstring{ Trobau el conjunt de punts que satisfà la següent inequació: $|x - |x - 2| < 3$}{|x - |x - 2| < 3}}

\begin{itemize}
    \item Casos pel valor absolut:
\end{itemize}

\[
|x - 2| =
\begin{cases}
x - 2 & \text{si } x \geq 2 \\
-x + 2 & \text{si } x < 2
\end{cases}
\]

\begin{itemize}
    \item $x \geq 2$
\end{itemize}
$|x-x+2| = |2| = 2 < 3$ Sempre

\begin{itemize}
    \item $x < 2$
\end{itemize}

$|x-(-x+2)| = |x+x-2| = |2x-2| < 3$ \\
$-3<2x-2<3 \Rightarrow -1 <2x<5 \Rightarrow \frac{-1}{2} < x < \frac{5}{2} \Rightarrow x \in (\frac{-1}{2}, \frac{5}{2})$ \\
Tenint en compte $x<2$: \fbox{$x \in (\frac{-1}{2}, 2)$}\\


\textbf{Solució: $x \in (\frac{-1}{2}, \infty)$}
